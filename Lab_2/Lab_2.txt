where pc.model in
(
select distinct pc.model
from labor_sql.pc,labor_sql.product 
where pc.model in
(
select distinct product.model from labor_sql.product
where maker in (
select distinct product.maker
from labor_sql.pc
where  maker in (select maker from product where type='PC')
and maker in (select maker from product where type='Printer'))
and type ="PC"
)
)
;

9.

Select
	ships.name, classes.numGuns, classes.bore, classes.displacement, classes.type, 
    classes.country,ships.launched,classes.class 
From
	labor_sql.ships
	Join labor_sql.classes On ships.class=classes.class
Where 
	Case WHEN classes.numGuns = 12 THEN 1 ELSE 0 END + 
	Case WHEN classes.bore = 16 THEN 1 ELSE 0 END +
	Case WHEN classes.displacement = 46000 THEN 1 ELSE 0 END +
	Case WHEN classes.type = 'bc' THEN 1 ELSE 0 END +
	Case WHEN ships.launched = 1941 THEN 1 ELSE 0 END +
	Case WHEN classes.class='North Carolina' THEN 1 ELSE 0 END +
	Case WHEN classes.country='GT.Britain'THEN 1 ELSE 0 END >= 4;
